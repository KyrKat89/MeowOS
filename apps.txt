<script>
(()=>{
  const boardEl = document.getElementById('board');
  const statusEl = document.getElementById('status');
  const resetBtn = document.getElementById('resetBtn');

  const WIN_LINES = [
    [0,1,2],[3,4,5],[6,7,8], // rows
    [0,3,6],[1,4,7],[2,5,8], // cols
    [0,4,8],[2,4,6]          // diags
  ];

  let board, current, gameOver, moves, nextStarter = 'X';

  function init(){
    board = Array(9).fill('');
    current = nextStarter;          // alternate who starts each round
    nextStarter = nextStarter === 'X' ? 'O' : 'X';
    gameOver = false;
    moves = 0;
    boardEl.innerHTML = '';
    for(let i=0;i<9;i++){
      const btn = document.createElement('button');
      btn.className = 'cell';
      btn.type = 'button';
      btn.setAttribute('role','gridcell');
      btn.setAttribute('aria-label','Empty');
      btn.dataset.idx = i;
      btn.tabIndex = 0;
      btn.addEventListener('click', onCellClick);
      btn.addEventListener('keydown', e=>{
        if(e.key === 'Enter' || e.key === ' '){ e.preventDefault(); btn.click(); }
      });
      boardEl.appendChild(btn);
    }
    updateStatus(`${current} starts, meow.`);
  }

  function onCellClick(e){
    const idx = +e.currentTarget.dataset.idx;
    if(gameOver || board[idx]) return;

    board[idx] = current;
    moves++;
    paintCell(idx, current);

    const win = checkWin();
    if(win){
      gameOver = true;
      highlightWin(win);
      updateStatus(`${current} wins! purrfect ðŸ¾`);
      return;
    }
    if(moves === 9){
      gameOver = true;
      updateStatus(`It's a draw, meow.`);
      return;
    }
    current = current === 'X' ? 'O' : 'X';
    updateStatus(`${current}'s turn, meow.`);
  }

  function paintCell(idx, mark){
    const cell = boardEl.children[idx];
    cell.textContent = mark;
    cell.classList.add('filled', mark.toLowerCase());
    cell.setAttribute('aria-label', mark);
  }

  function updateStatus(text){
    statusEl.textContent = text;
  }

  function checkWin(){
    for(const line of WIN_LINES){
      const [a,b,c] = line;
      if(board[a] && board[a]===board[b] && board[a]===board[c]) return line;
    }
    return null;
  }

  function highlightWin(line){
    line.forEach(i => boardEl.children[i].classList.add('win'));
  }

  resetBtn.addEventListener('click', init);

  init();
})();
</script>

    const input = document.getElementById('url');

    function parseTimeToSeconds(t){
      if(!t) return 0;
      // Accept formats like 90, 1m30s, 2h10m, etc.
      if(/^\d+$/.test(t)) return parseInt(t,10);
      let total = 0; const re = /(\d+)(h|m|s)/g; let m;
      while((m = re.exec(t))){
        const val = parseInt(m[1],10);
        if(m[2] === 'h') total += val*3600;
        if(m[2] === 'm') total += val*60;
        if(m[2] === 's') total += val;
      }
      return total;
    }

    function extractVideoId(raw){
      try{
        const u = new URL(raw.trim());
        const host = u.hostname.replace(/^www\./,'');
        // watch?v=ID
        if(host.endsWith('youtube.com')){
          // Shorts: /shorts/ID
          const shortsMatch = u.pathname.match(/^\/shorts\/([a-zA-Z0-9_-]{6,})/);
          if(shortsMatch) return { id: shortsMatch[1], t: u.searchParams.get('t') };
          // Standard watch
          const v = u.searchParams.get('v');
          if(v) return { id: v, t: u.searchParams.get('t') };
          // /embed/ID
          const em = u.pathname.match(/^\/embed\/([a-zA-Z0-9_-]{6,})/);
          if(em) return { id: em[1], t: u.searchParams.get('t') };
        }
        if(host === 'youtu.be'){
          const m = u.pathname.match(/^\/([a-zA-Z0-9_-]{6,})/);
          if(m) return { id: m[1], t: u.searchParams.get('t') };
        }
      }catch(e){ /* fall through */ }
      // If raw looks like an id already
      if(/^[a-zA-Z0-9_-]{6,}$/.test(raw.trim())) return { id: raw.trim(), t: null };
      return null;
    }

    function play(){
      const val = input.value;
      const res = extractVideoId(val);
      if(!res){
        errorEl.style.display = 'block';
        return;
      }
      errorEl.style.display = 'none';
      const start = parseTimeToSeconds(res.t);
      const params = new URLSearchParams({autoplay:1, modestbranding:1, rel:0, iv_load_policy:3});
      if(start>0) params.set('start', String(start));
      // Use origin if available (helps with some browsers)
      try{ params.set('origin', location.origin); }catch{}
      const src = `https://www.youtube-nocookie.com/embed/${encodeURIComponent(res.id)}?${params.toString()}`;
      iframe.src = src;
      // Move focus to player for keyboard shortcuts
      iframe.focus({preventScroll:true});
    }
